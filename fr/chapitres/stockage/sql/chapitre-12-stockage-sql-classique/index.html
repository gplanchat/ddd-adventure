    class Payment
{
    private function __construct(
        private PaymentId $id,
        private Money $amount,
        private PaymentStatus $status,
        private ?DateTime $processedAt = null
    ) {}

    public static function create(PaymentId $id, Money $amount): self
    {
        if ($amount->isNegative()) {
            throw new InvalidAmountException('Payment amount cannot be negative');
        }
        
        return new self($id, $amount, PaymentStatus::PENDING);
    }

    public function process(Money $amount): void
    {
        if (!$this->canBeProcessed()) {
            throw new PaymentCannotBeProcessedException();
        }
        
        if ($amount->isGreaterThan($this->getMaxAllowedAmount())) {
            throw new AmountExceedsMaximumException();
        }
        
        $this->amount = $amount;
        $this->status = PaymentStatus::PROCESSED;
        $this->processedAt = new DateTime();
    }

    private function canBeProcessed(): bool
    {
        return $this->status === PaymentStatus::PENDING;
    }

    private function getMaxAllowedAmount(): Money
    {
        return $this->amount->multiply(1.1); // 10% de marge
    }

    // Getters publics
    public function getId(): PaymentId { return $this->id; }
    public function getAmount(): Money { return $this->amount; }
    public function getStatus(): PaymentStatus { return $this->status; }
    public function getProcessedAt(): ?DateTime { return $this->processedAt; }
}

    class DoctrinePaymentRepository implements PaymentRepository
{
    public function __construct(
        private EntityManagerInterface $entityManager
    ) {}

    public function save(Payment $payment): void
    {
        $doctrineEntity = $this->convertToDoctrineEntity($payment);
        $this->entityManager->persist($doctrineEntity);
        $this->entityManager->flush();
    }

    public function find(PaymentId $id): ?Payment
    {
        $doctrineEntity = $this->entityManager->find(Payment::class, $id->toString());
        
        if (!$doctrineEntity) {
            return null;
        }
        
        return $this->convertToDomainEntity($doctrineEntity);
    }

    public function findByStatus(PaymentStatus $status): Collection
    {
        $qb = $this->entityManager->createQueryBuilder();
        $qb->select('p')
           ->from(Payment::class, 'p')
           ->where('p.status = :status')
           ->setParameter('status', $status->value);

        $doctrineEntities = $qb->getQuery()->getResult();
        
        return new ArrayCollection(
            array_map([$this, 'convertToDomainEntity'], $doctrineEntities)
        );
    }

    public function remove(Payment $payment): void
    {
        $doctrineEntity = $this->convertToDoctrineEntity($payment);
        $this->entityManager->remove($doctrineEntity);
        $this->entityManager->flush();
    }

    private function convertToDoctrineEntity(Payment $payment): Payment
    {
        $doctrineEntity = new Payment();
        $doctrineEntity->setId($payment->getId()->toString());
        $doctrineEntity->setAmount($payment->getAmount()->getAmount());
        $doctrineEntity->setCurrency($payment->getAmount()->getCurrency()->value);
        $doctrineEntity->setStatus($payment->getStatus()->value);
        $doctrineEntity->setProcessedAt($payment->getProcessedAt());
        
        return $doctrineEntity;
    }

    private function convertToDomainEntity(Payment $doctrineEntity): Payment
    {
        return Payment::fromExisting(
            new PaymentId($doctrineEntity->getId()),
            new Money($doctrineEntity->getAmount(), Currency::from($doctrineEntity->getCurrency())),
            PaymentStatus::from($doctrineEntity->getStatus()),
            $doctrineEntity->getProcessedAt()
        );
    }
}

    class PaymentProvider implements ProviderInterface
{
    public function __construct(
        private PaymentRepository $repository
    ) {}

    public function provide(Operation $operation, array $uriVariables = [], array $context = []): object|array|null
    {
        if ($operation->getName() === 'get') {
            $payment = $this->repository->find(new PaymentId($uriVariables['id']));
            return $payment ? Payment::fromDomain($payment) : null;
        }

        if ($operation->getName() === 'get_collection') {
            $payments = $this->repository->findAll();
            return array_map([Payment::class, 'fromDomain'], $payments->toArray());
        }

        return null;
    }
}

    class PaymentService
{
    public function __construct(
        private PaymentRepository $repository,
        private EntityManagerInterface $entityManager
    ) {}

    public function processPayment(PaymentId $id, Money $amount): Payment
    {
        return $this->entityManager->transactional(function() use ($id, $amount) {
            $payment = $this->repository->find($id);
            
            if (!$payment) {
                throw new PaymentNotFoundException();
            }
            
            $payment->process($amount);
            $this->repository->save($payment);
            
            return $payment;
        });
    }
}
