    class EventSourcedRepository implements CommandRepository
{
    public function __construct(
        private EventStore $eventStore,
        private EventBus $eventBus
    ) {}

    public function save(AggregateRoot $aggregate): void
    {
        $events = $aggregate->getEvents();
        
        if (empty($events)) {
            return;
        }
        
        $this->eventStore->append(
            $aggregate->getId(),
            $events,
            $aggregate->getVersion()
        );
        
        // Publier les √©v√©nements
        foreach ($events as $event) {
            $this->eventBus->publish($event);
        }
    }

    public function find(Id $id): ?AggregateRoot
    {
        $events = $this->eventStore->getEvents($id);
        
        if (empty($events)) {
            return null;
        }
        
        return $this->reconstituteFromEvents($events);
    }

    private function reconstituteFromEvents(array $events): AggregateRoot
    {
        $aggregate = $this->createEmptyAggregate();
        
        foreach ($events as $event) {
            $aggregate->apply($event);
        }
        
        return $aggregate;
    }
}
